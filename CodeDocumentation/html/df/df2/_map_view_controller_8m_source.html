00162 -(void)zoomToFitMapAnnotations:(MKMapView*)mapView
00163 {
00164     if([mapView.annotations count] == 0)
00165         return;
00166     
00167     CLLocationCoordinate2D topLeftCoord;
00168     topLeftCoord.latitude = -90;
00169     topLeftCoord.longitude = 180;
00170     
00171     CLLocationCoordinate2D bottomRightCoord;
00172     bottomRightCoord.latitude = 90;
00173     bottomRightCoord.longitude = -180;
00174     
00175     for(MapLocations *annotation in mapView.annotations)
00176     {
00177         topLeftCoord.longitude = fmin(topLeftCoord.longitude, annotation.coordinate.longitude);
00178         topLeftCoord.latitude = fmax(topLeftCoord.latitude, annotation.coordinate.latitude);
00179         
00180         bottomRightCoord.longitude = fmax(bottomRightCoord.longitude, annotation.coordinate.longitude);
00181         bottomRightCoord.latitude = fmin(bottomRightCoord.latitude, annotation.coordinate.latitude);
00182     }
00183     
00184     MKCoordinateRegion region;
00185     region.center.latitude = topLeftCoord.latitude - (topLeftCoord.latitude - bottomRightCoord.latitude) * 0.5;
00186     region.center.longitude = topLeftCoord.longitude + (bottomRightCoord.longitude - topLeftCoord.longitude) * 0.5;
00187     region.span.latitudeDelta = fabs(topLeftCoord.latitude - bottomRightCoord.latitude) * 1.5; // Add a little extra space on the sides
00188     region.span.longitudeDelta = fabs(bottomRightCoord.longitude - topLeftCoord.longitude) * 1.5; // Add a little extra space on the sides
00189     
00190     region = [mapView regionThatFits:region];
00191     [mapView setRegion:region animated:YES];
